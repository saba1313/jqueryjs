This example utilizes the advanced callback onDrag, which gets the position as parameters and can return these to i.e. make a unique containment.

$("#mydiv").drag({
helper: 'original',
onDrag: function(helper,x,y)
{
    var centerx = 800;
    var centery = 400;
    var angle = Math.atan((centery-y)/(centerx-x));
    if((centerx-x)>=0)
        angle += Math.PI;
    radius = 160;
    return {
        x: radius * Math.cos(angle) + centerx,
        y: radius * Math.sin(angle) + centery
    }
}
});
