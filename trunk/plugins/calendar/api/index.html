Calendar Settings

Setting Explanation
clearText
  • @type: String [text]
  • @desc: The text to display for the clear link.
  • @example: 
    clearText = 'Clear';
closeText
  • @type: String [text]
  • @desc: The text to display for the close link.
  • @example: 
    closeText = 'Close';
prevText
  • @type: String [text]
  • @desc: The text to display for the previous month link.
  • @example: 
    prevText = '&lt;Prev';
nextText
  • @type: String [text]
  • @desc: The text to display for the next month link.
  • @example: 
    nextText = 'Next&gt;';
currentText
  • @type: String [text]
  • @desc: The text to display for the current month link.
  • @example: 
    currentText = 'Today';
dayNames
  • @type: Array [DayNames]
  • @desc: The list of day names starting from Sunday.
  • @example: 
    dayNames = new Array('Sun','Mon','Tue','Wed','Thu','Fri','Sat');
monthNames
  • @type: Array [MonthNames]
  • @desc: The list of month names.
  • @example: 
    monthNames = new Array('Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec');
dateFormat
  • @type: String [format]
  • @desc: The format for parsed and displayed dates - first three characters are day, month, year in the required order, fourth is the separator, e.g. US would be 'MDY/'
  • @example: 
    dateFormat = 'MDY/';
appendText
  • @type: String [text]
  • @desc: The text to display after each date field, e.g. to show the required format.
  • @example: 
    appendText = ' (yyyy-mm-dd)';
speed
  • @type: String [slow, medium or fast] or Number [milliseconds]
  • @desc: Control the speed at which the calendar appears, it may be a time in milliseconds, a descriptive name ('slow', 'medium', 'fast'), or '' for immediately.
  • @example: 
    speed = 'medium';
firstDay
  • @type: Number [DayofWeek]
  • @desc: Set the first day of the week, Sunday = 0, Monday = 1, ...
  • @example: 
    firstDay = 0;
yearRange
  • @type: String [-int:+int] or ['yyyy:yyyy']
  • @desc: Control the range of years displayed in the year drop-down, either relative to current year (-nn:+nn) or absolute (nnnn:nnnn), e.g. '2000:2020'.
  • @example: 
    yearRange = '-10:+10';
minDate
  • @type: Date Object [Date(yyyy, m - 1, dd)]
  • @desc: Set a minimum selectable date, or null for no limit.
  • @example: 
    minDate = new Date(2000, 7 - 1, 1); // 01-Jul-2000
maxDate
  • @type: Date Object [Date(yyyy, m - 1, dd)]
  • @desc: Set a maximum selectable date, or null for no limit.
  • @example: 
    maxDate = new Date(2009, 6 - 1, 30); // 30-Jun-2009
autoPopUp
  • @type: String (focus, button or both)
  • @desc: Have the calendar appear automatically when the field receives focus (focus), appear only when a button is clicked (button), or appear when either event takes place (both).
  • @example: 
    autoPopUp = true;
buttonText
  • @type: String [text]
  • @desc: Use the following in conjunction with autoPopUp = false The text to display on the popup button.
  • @example: 
    buttonText = 'Calendar';
buttonImage
  • @type: String (Path)
  • @desc: The URL for the popup button image, if set button text becomes the alt value and is not directly displayed.
  • @example: 
    buttonImage = '/images/calendar.gif';
buttonImageOnly
  • @type: Bool (True or False)
  • @desc: Place an image after the field without a button.
  • @example: 
    buttonImageOnly = true;
closeAtTop
  • @type: Bool (True or False)
  • @desc: Position the clear/close links at the top or bottom.
  • @example: 
    closeAtTop = true;
showOtherMonths
  • @type: Bool (True or False)
  • @desc: Display dates in other months (non-selectable).
  • @example: 
    showOtherMonths = true;
customDate
  • @type: Function
  • @desc: The function takes a date as a parameter and must return an array with [0] = true/false indicating whether or not this date is selectable and [1] = CSS class name(s) or '' for custom presentation.
  • @example: 
    customDate = nationalDays;
								
natDays = [[1, 26, 'au'], [2, 6, 'nz'], [3, 17, 'ie'], [4, 27, 'za'], [5, 25, 'ar'], [6, 6, 'se'],
			[7, 4, 'us'], [8, 17, 'id'], [9, 7, 'br'], [10, 1, 'cn'], [11, 22, 'lb'], [12, 12, 'ke']];
function nationalDays(date) {
	for (i = 0; i < natDays.length; i++) {
		if (date.getMonth() == natDays[i][0] - 1 && date.getDate() == natDays[i][1]) {
			return [false, natDays[i][2] + '_day'];
		}
	}
	return [true, ''];
}
  • @note: One built in function of this type exists called "noWeekends".
  • @example: 
    customDate = popUpCal.noWeekends;
fieldSettings
  • @type: Function
  • @desc: Can be a function that takes an input field and returns a settings(anonymous) object to update the calendar with.
  • @example: 
    function customSettings(input) {
	range = (input.className == 'calendarFocus' ? 5 : 
		(input.className == 'calendarButton' ? 7 : 10));
	return {yearRange: '-' + range + ':+' + range, 
		minDate: (input.id == 'dTo' ? getDate($('#dFrom')) : defaultMinDate), 
		maxDate: (input.id == 'dFrom' ? getDate($('#dTo')) : defaultMaxDate)};
}
    Then in the initialize function add: 
    $('.calendarFocus').calendar({appendText: ' (default format dd/mm/yyyy)'});
$('.calendarButton').calendar({appendText: ' ', autoPopUp: 'button'});
$('.calendarBoth').calendar({autoPopUp: 'both', buttonImageOnly: true, 
	buttonImage: 'img/calendar.gif', buttonText: 'Calendar'});
$('.calendarRange').calendar(); // Same as above
changeFirstDay
  • @type: Bool (True or False)
  • @desc: Allows you to click on the day names to have the week start on that day. You can disable this feature by setting the changeFirstDay attribute to false.
  • @example: 
    changeFirstDay = false;
inline
  • @type: bool
  • @desc: Allows you to show the calendar at all times. You must attach the calendar to a div in order for it to work. NOTE:This does not work with multiple calendars on the page).
  • @example: 
    inline = true;
customEvent
  • @type: Function
  • @desc: Allows you to define your own event when the calendar is selected. Use this with inline calendar.
  • @example: 
    customEvent:function () {
	alert(this.date);
}
